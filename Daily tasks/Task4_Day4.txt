  localJSON = { whoCanJoin: "CAN_REQUEST_TO_JOIN", allowExternalMembers: "false"};

function compareObjects(object1, object2) {
  for (i in object1){
    if (!object2.hasOwnProperty(i))
      return false;
  return true;
  }
}
